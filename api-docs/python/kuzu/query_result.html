180    def get_as_networkx(self, directed=True):
181        """
182        Convert the nodes and rels in query result into a NetworkX directed or undirected graph
183        with the following rules:
184        Columns with data type other than node or rel will be ignored.
185        Duplicated nodes and rels will be converted only once.
186
187        Parameters
188        ----------
189        directed : bool
190            Whether the graph should be directed. Defaults to True.
191
192        Returns
193        -------
194        networkx.DiGraph or networkx.Graph
195            Query result as a NetworkX graph.
196
197        """
198
199        self.check_for_query_result_close()
200        import networkx as nx
201
202        if directed:
203            nx_graph = nx.DiGraph()
204        else:
205            nx_graph = nx.Graph()
206        properties_to_extract = self._get_properties_to_extract()
207
208        self.reset_iterator()
209
210        nodes = {}
211        rels = {}
212        table_to_label_dict = {}
213        table_primary_key_dict = {}
214
215        def encode_node_id(node, table_primary_key_dict):
216            return node['_label'] + "_" + str(node[table_primary_key_dict[node['_label']]])
217
218        # De-duplicate nodes and rels
219        while self.has_next():
220            row = self.get_next()
221            for i in properties_to_extract:
222                column_type, _ = properties_to_extract[i]
223                if column_type == Type.NODE.value:
224                    _id = row[i]["_id"]
225                    nodes[(_id["table"], _id["offset"])] = row[i]
226                    table_to_label_dict[_id["table"]] = row[i]["_label"]
227
228                elif column_type == Type.REL.value:
229                    _src = row[i]["_src"]
230                    _dst = row[i]["_dst"]
231                    rels[(_src["table"], _src["offset"], _dst["table"],
232                          _dst["offset"])] = row[i]
233
234                elif column_type == Type.RECURSIVE_REL.value:
235                    for node in row[i]['_nodes']:
236                        _id = node["_id"]
237                        nodes[(_id["table"], _id["offset"])] = node
238                        table_to_label_dict[_id["table"]] = node["_label"]
239                    for rel in row[i]['_rels']:
240                        for key in rel:
241                            if rel[key] is None:
242                                del rel[key]
243                        _src = rel["_src"]
244                        _dst = rel["_dst"]
245                        rels[(_src["table"], _src["offset"], _dst["table"],
246                              _dst["offset"])] = rel
247
248        # Add nodes
249        for node in nodes.values():
250            _id = node["_id"]
251            node_id = node['_label'] + "_" + str(_id["offset"])
252            if node['_label'] not in table_primary_key_dict:
253                props = self.connection._get_node_property_names(
254                    node['_label'])
255                for prop_name in props:
256                    if props[prop_name]['is_primary_key']:
257                        table_primary_key_dict[node['_label']] = prop_name
258                        break
259            node_id = encode_node_id(node, table_primary_key_dict)
260            node[node['_label']] = True
261            nx_graph.add_node(node_id, **node)
262
263        # Add rels
264        for rel in rels.values():
265            _src = rel["_src"]
266            _dst = rel["_dst"]
267            src_node = nodes[(_src["table"], _src["offset"])]
268            dst_node = nodes[(_dst["table"], _dst["offset"])]
269            src_id = encode_node_id(src_node, table_primary_key_dict)
270            dst_id = encode_node_id(dst_node, table_primary_key_dict)
271            nx_graph.add_edge(src_id, dst_id, **rel)
272        return nx_graph

274    def _get_properties_to_extract(self):
275        column_names = self.get_column_names()
276        column_types = self.get_column_data_types()
277        properties_to_extract = {}
278
279        # Iterate over columns and extract nodes and rels, ignoring other columns
280        for i in range(len(column_names)):
281            column_name = column_names[i]
282            column_type = column_types[i]
283            if column_type in [Type.NODE.value, Type.REL.value, Type.RECURSIVE_REL.value]:
284                properties_to_extract[i] = (column_type, column_name)
285        return properties_to_extract

287    def get_as_torch_geometric(self):
288        """
289        Converts the nodes and rels in query result into a PyTorch Geometric graph representation
290        torch_geometric.data.Data or torch_geometric.data.HeteroData.
291
292        For node conversion, numerical and boolean properties are directly converted into tensor and
293        stored in Data/HeteroData. For properties cannot be converted into tensor automatically 
294        (please refer to the notes below for more detail), they are returned as unconverted_properties.
295
296        For rel conversion, rel is converted into edge_index tensor director. Edge properties are returned
297        as edge_properties.
298
299        Node properties that cannot be converted into tensor automatically:
300        - If the type of a node property is not one of INT64, DOUBLE, or BOOL, it cannot be converted
301          automatically.
302        - If a node property contains a null value, it cannot be converted automatically.
303        - If a node property contains a nested list of variable length (e.g. [[1,2],[3]]), it cannot be 
304          converted automatically.
305        - If a node property is a list or nested list, but the shape is inconsistent (e.g. the list length 
306          is 6 for one node but 5 for another node), it cannot be converted automatically.
307
308        Additional conversion rules:
309        - Columns with data type other than node or rel will be ignored.
310        - Duplicated nodes and rels will be converted only once.
311
312        Returns
313        -------
314        torch_geometric.data.Data or torch_geometric.data.HeteroData
315            Query result as a PyTorch Geometric graph. Containing numeric or boolean node properties
316            and edge_index tensor.
317
318        dict
319            A dictionary that maps the positional offset of each node in Data/HeteroData to its primary
320            key in the database.
321
322        dict
323            A dictionary contains node properties that cannot be converted into tensor automatically. The
324            order of values for each property is aligned with nodes in Data/HeteroData.
325
326        dict
327            A dictionary contains edge properties. The order of values for each property is aligned with
328            edge_index in Data/HeteroData.
329        """
330
331        self.check_for_query_result_close()
332        # Despite we are not using torch_geometric in this file, we need to
333        # import it here to throw an error early if the user does not have
334        # torch_geometric or torch installed.
335
336        import torch
337        import torch_geometric

339        converter = TorchGeometricResultConverter(self)
340        return converter.get_as_torch_geometric()

def get_as_networkx(self, directed=True): 
181    def get_as_networkx(self, directed=True):
182        """
183        Convert the nodes and rels in query result into a NetworkX directed or undirected graph
184        with the following rules:
185        Columns with data type other than node or rel will be ignored.
186        Duplicated nodes and rels will be converted only once.
187
188        Parameters
189        ----------
190        directed : bool
191            Whether the graph should be directed. Defaults to True.
192
193        Returns
194        -------
195        networkx.DiGraph or networkx.Graph
196            Query result as a NetworkX graph.
197
198        """
199
200        self.check_for_query_result_close()
201        import networkx as nx
202
203        if directed:
204            nx_graph = nx.DiGraph()
205        else:
206            nx_graph = nx.Graph()
207        properties_to_extract = self._get_properties_to_extract()
208
209        self.reset_iterator()
210
211        nodes = {}
212        rels = {}
213        table_to_label_dict = {}
214        table_primary_key_dict = {}
215
216        def encode_node_id(node, table_primary_key_dict):
217            return node['_label'] + "_" + str(node[table_primary_key_dict[node['_label']]])
218
219        # De-duplicate nodes and rels
220        while self.has_next():
221            row = self.get_next()
222            for i in properties_to_extract:
223                column_type, _ = properties_to_extract[i]
224                if column_type == Type.NODE.value:
225                    _id = row[i]["_id"]
226                    nodes[(_id["table"], _id["offset"])] = row[i]
227                    table_to_label_dict[_id["table"]] = row[i]["_label"]
228
229                elif column_type == Type.REL.value:
230                    _src = row[i]["_src"]
231                    _dst = row[i]["_dst"]
232                    rels[(_src["table"], _src["offset"], _dst["table"],
233                          _dst["offset"])] = row[i]
234
235                elif column_type == Type.RECURSIVE_REL.value:
236                    for node in row[i]['_nodes']:
237                        _id = node["_id"]
238                        nodes[(_id["table"], _id["offset"])] = node
239                        table_to_label_dict[_id["table"]] = node["_label"]
240                    for rel in row[i]['_rels']:
241                        for key in rel:
242                            if rel[key] is None:
243                                del rel[key]
244                        _src = rel["_src"]
245                        _dst = rel["_dst"]
246                        rels[(_src["table"], _src["offset"], _dst["table"],
247                              _dst["offset"])] = rel
248
249        # Add nodes
250        for node in nodes.values():
251            _id = node["_id"]
252            node_id = node['_label'] + "_" + str(_id["offset"])
253            if node['_label'] not in table_primary_key_dict:
254                props = self.connection._get_node_property_names(
255                    node['_label'])
256                for prop_name in props:
257                    if props[prop_name]['is_primary_key']:
258                        table_primary_key_dict[node['_label']] = prop_name
259                        break
260            node_id = encode_node_id(node, table_primary_key_dict)
261            node[node['_label']] = True
262            nx_graph.add_node(node_id, **node)
263
264        # Add rels
265        for rel in rels.values():
266            _src = rel["_src"]
267            _dst = rel["_dst"]
268            src_node = nodes[(_src["table"], _src["offset"])]
269            dst_node = nodes[(_dst["table"], _dst["offset"])]
270            src_id = encode_node_id(src_node, table_primary_key_dict)
271            dst_id = encode_node_id(dst_node, table_primary_key_dict)
272            nx_graph.add_edge(src_id, dst_id, **rel)
273        return nx_graph

Converts the nodes and rels in query result into a PyTorch Geometric graph representation torch_geometric.data.Data or torch_geometric.data.HeteroData.

For node conversion, numerical and boolean properties are directly converted into tensor and stored in Data/HeteroData. For properties cannot be converted into tensor automatically (please refer to the notes below for more detail), they are returned as unconverted_properties.

For rel conversion, rel is converted into edge_index tensor director. Edge properties are returned as edge_properties.

Node properties that cannot be converted into tensor automatically:

  • If the type of a node property is not one of INT64, DOUBLE, or BOOL, it cannot be converted automatically.
  • If a node property contains a null value, it cannot be converted automatically.
  • If a node property contains a nested list of variable length (e.g. [[1,2],[3]]), it cannot be converted automatically.
  • If a node property is a list or nested list, but the shape is inconsistent (e.g. the list length is 6 for one node but 5 for another node), it cannot be converted automatically.

Additional conversion rules:

  • Columns with data type other than node or rel will be ignored.
  • Duplicated nodes and rels will be converted only once.
