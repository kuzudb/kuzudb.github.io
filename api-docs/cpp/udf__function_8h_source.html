160 template<typename TR, typename... Args>
161 static scalar_exec_func getScalarExecFunc(
162 TR (*udfFunc)(Args...), std::vector<common::LogicalTypeID> parameterTypes) {
163 constexpr auto numArgs = sizeof...(Args);
164 switch (numArgs) {
165 case 1:
166 return createUnaryExecFunc<TR, Args...>(udfFunc, std::move(parameterTypes));
167 case 2:
168 return createBinaryExecFunc<TR, Args...>(udfFunc, std::move(parameterTypes));
169 case 3:
170 return createTernaryExecFunc<TR, Args...>(udfFunc, std::move(parameterTypes));
171 default:
172 throw common::BinderException("UDF function only supported until ternary!");
173 }
174 }
