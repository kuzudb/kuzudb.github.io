53/*
54 * The Memory Manager (MM) is used for allocating/reclaiming intermediate memory blocks.
55 * It can allocate a memory buffer of size PAGE_256KB from the buffer manager backed by a
56 * BMFileHandle with temp in-mem file.
57 *
58 * Internally, MM uses a MemoryAllocator. The MemoryAllocator is holding the BMFileHandle backed by
59 * a temp in-mem file, and responsible for allocating/reclaiming memory buffers of its size class
60 * from the buffer manager. The MemoryAllocator keeps track of free pages in the BMFileHandle, so
61 * that it can reuse those freed pages without allocating new pages. The MemoryAllocator is
62 * thread-safe, so that multiple threads can allocate/reclaim memory blocks with the same size class
63 * at the same time.
64 *
65 * MM will return a MemoryBuffer to the caller, which is a wrapper of the allocated memory block,
66 * and it will automatically call its allocator to reclaim the memory block when it is destroyed.
67 */
68class MemoryManager {
69public:
70 explicit MemoryManager(BufferManager* bm, common::VirtualFileSystem* vfs) : bm{bm} {
71 allocator = std::make_unique<MemoryAllocator>(bm, vfs);
72 }
73
74 inline std::unique_ptr<MemoryBuffer> allocateBuffer(bool initializeToZero = false) {
75 return allocator->allocateBuffer(initializeToZero);
76 }
77 inline BufferManager* getBufferManager() const { return bm; }
78
79private:
80 BufferManager* bm;
81 std::unique_ptr<MemoryAllocator> allocator;
