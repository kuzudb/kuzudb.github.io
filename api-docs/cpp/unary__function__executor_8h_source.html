110 template<typename OPERAND_TYPE, typename RESULT_TYPE, typename FUNC, typename OP_WRAPPER>
111 static void executeSwitch(common::ValueVector& operand, common::ValueVector& result,
112 void* dataPtr) {
113 result.resetAuxiliaryBuffer();
114 auto& operandSelVector = operand.state->getSelVector();
115 if (operand.state->isFlat()) {
116 auto inputPos = operandSelVector[0];
117 auto resultPos = result.state->getSelVector()[0];
118 result.setNull(resultPos, operand.isNull(inputPos));
119 if (!result.isNull(resultPos)) {
120 executeOnValue<OPERAND_TYPE, RESULT_TYPE, FUNC, OP_WRAPPER>(operand, inputPos,
121 result, resultPos, dataPtr);
122 }
123 } else {
124 if (operand.hasNoNullsGuarantee()) {
125 result.setAllNonNull();
126 if (operandSelVector.isUnfiltered()) {
127 for (auto i = 0u; i < operandSelVector.getSelSize(); i++) {
128 executeOnValue<OPERAND_TYPE, RESULT_TYPE, FUNC, OP_WRAPPER>(operand, i,
129 result, i, dataPtr);
130 }
131 } else {
132 for (auto i = 0u; i < operandSelVector.getSelSize(); i++) {
133 auto pos = operandSelVector[i];
134 executeOnValue<OPERAND_TYPE, RESULT_TYPE, FUNC, OP_WRAPPER>(operand, pos,
135 result, pos, dataPtr);
136 }
137 }
138 } else {
139 if (operandSelVector.isUnfiltered()) {
140 for (auto i = 0u; i < operandSelVector.getSelSize(); i++) {
141 result.setNull(i, operand.isNull(i));
142 if (!result.isNull(i)) {
143 executeOnValue<OPERAND_TYPE, RESULT_TYPE, FUNC, OP_WRAPPER>(operand, i,
144 result, i, dataPtr);
145 }
146 }
147 } else {
148 for (auto i = 0u; i < operandSelVector.getSelSize(); i++) {
149 auto pos = operandSelVector[i];
150 result.setNull(pos, operand.isNull(pos));
151 if (!result.isNull(pos)) {
152 executeOnValue<OPERAND_TYPE, RESULT_TYPE, FUNC, OP_WRAPPER>(operand,
153 pos, result, pos, dataPtr);
154 }
155 }
156 }
157 }
158 }
159 }
