25// Compression type is written to the data header both so we can usually catch issues when we
26// decompress uncompressed data by mistake, and to allow for runtime-configurable compression.
27enum class CompressionType : uint8_t {
28 UNCOMPRESSED = 0,
29 INTEGER_BITPACKING = 1,
30 BOOLEAN_BITPACKING = 2,
31 CONSTANT = 3,
32};

34struct CompressionMetadata {

56class CompressionAlg {
57public:
58 virtual ~CompressionAlg() = default;
59
60 // Takes a single uncompressed value from the srcBuffer and compresses it into the dstBuffer
61 // Offsets refer to value offsets, not byte offsets
62 //
63 // nullMask may be null if no mask is available (all values are non-null)
64 // Storage of null values is handled by the implementation and decompression of null values
65 // does not have to produce the original value passed to this function.
66 virtual void setValuesFromUncompressed(const uint8_t* srcBuffer, common::offset_t srcOffset,
67 uint8_t* dstBuffer, common::offset_t dstOffset, common::offset_t numValues,
68 const CompressionMetadata& metadata, const common::NullMask* nullMask) const = 0;
69
70 // Returns compression metadata, including any relevant parameters specific to this dataset
71 // which will need to be passed to compressNextPage. Since this may need to scan the entire
72 // buffer, which is slow, it should only be called once for each set of data to compress.
73 virtual CompressionMetadata getCompressionMetadata(const uint8_t* srcBuffer,
74 uint64_t numValues) const = 0;
75
76 // Takes uncompressed data from the srcBuffer and compresses it into the dstBuffer
77 //
78 // stores only as much data in dstBuffer as will fit, and advances the srcBuffer pointer
79 // to the beginning of the next value to store.
80 // (This means that we can't start the next page on an unaligned value.
81 // Maybe instead we could use value offsets, but the compression algorithms
82 // usually work on aligned chunks anyway)
83 //
84 // dstBufferSize is the size in bytes
85 // numValuesRemaining is the number of values remaining in the srcBuffer to be compressed.
86 // compressNextPage must store the least of either the number of values per page
87 // (as calculated by CompressionMetadata::numValues), or the remaining number of values.
88 //
89 // returns the size in bytes of the compressed data within the page (rounded up to the nearest
90 // byte)
91 virtual uint64_t compressNextPage(const uint8_t*& srcBuffer, uint64_t numValuesRemaining,
92 uint8_t* dstBuffer, uint64_t dstBufferSize,
93 const struct CompressionMetadata& metadata) const = 0;
94
95 // Takes compressed data from the srcBuffer and decompresses it into the dstBuffer
96 // Offsets refer to value offsets, not byte offsets
97 // srcBuffer points to the beginning of a page
98 virtual void decompressFromPage(const uint8_t* srcBuffer, uint64_t srcOffset,
99 uint8_t* dstBuffer, uint64_t dstOffset, uint64_t numValues,
100 const CompressionMetadata& metadata) const = 0;
101};

145class Uncompressed : public CompressionAlg {
146public:

148 : numBytesPerValue{getDataTypeSizeInChunk(logicalType)} {}

152
153 inline void setValuesFromUncompressed(const uint8_t* srcBuffer, common::offset_t srcOffset,
154 uint8_t* dstBuffer, common::offset_t dstOffset, common::offset_t numValues,
155 const CompressionMetadata& /*metadata*/, const common::NullMask* /*nullMask*/) const final {
156 memcpy(dstBuffer + dstOffset * numBytesPerValue, srcBuffer + srcOffset * numBytesPerValue,
157 numBytesPerValue * numValues);
158 }

160 static inline uint64_t numValues(uint64_t dataSize, const common::LogicalType& logicalType) {
161 auto numBytesPerValue = getDataTypeSizeInChunk(logicalType);
162 return numBytesPerValue == 0 ? UINT64_MAX : dataSize / numBytesPerValue;
163 }

165 inline CompressionMetadata getCompressionMetadata(const uint8_t* /*srcBuffer*/,
166 uint64_t /*numValues*/) const override {
167 return CompressionMetadata();
168 }
169
170 inline uint64_t compressNextPage(const uint8_t*& srcBuffer, uint64_t numValuesRemaining,
171 uint8_t* dstBuffer, uint64_t dstBufferSize,
172 const struct CompressionMetadata& /*metadata*/) const override {
173 if (numBytesPerValue == 0) {
174 return 0;
175 }
176 uint64_t numValues = std::min(numValuesRemaining, dstBufferSize / numBytesPerValue);
177 uint64_t sizeToCopy = numValues * numBytesPerValue;
178 KU_ASSERT(sizeToCopy <= dstBufferSize);
179 std::memcpy(dstBuffer, srcBuffer, sizeToCopy);
180 srcBuffer += sizeToCopy;
181 return sizeToCopy;
182 }
183
184 inline void decompressFromPage(const uint8_t* srcBuffer, uint64_t srcOffset, uint8_t* dstBuffer,
185 uint64_t dstOffset, uint64_t numValues,
186 const CompressionMetadata& /*metadata*/) const override {
187 std::memcpy(dstBuffer + dstOffset * numBytesPerValue,
188 srcBuffer + srcOffset * numBytesPerValue, numValues * numBytesPerValue);
189 }
190
191protected:
192 const uint32_t numBytesPerValue;
193};

195// Serialized as nine bytes.
196// In the first byte:
197// Six bits are needed for the bit width (fewer for smaller types, but the header byte is the same
198// for simplicity)
199// One bit (the eighth) is needed to indicate if there are negative values
200// The seventh bit is unused
201// The eight remaining bytes are used for the offset
202struct BitpackHeader {
203 uint8_t bitWidth;
204 bool hasNegative;
205 // Offset to apply to all values
206 // Stored as unsigned, but for signed variants of IntegerBitpacking
207 // it gets cast to a signed type on use, letting it also store negative offsets
208 uint64_t offset;
209 static const uint8_t NEGATIVE_FLAG = 0b10000000;
210 static const uint8_t BITWIDTH_MASK = 0b01111111;
211
212 std::array<uint8_t, CompressionMetadata::DATA_SIZE> getData() const;

219template<typename T>
220class IntegerBitpacking : public CompressionAlg {
221 using U = std::make_unsigned_t<T>;
222 // This is an implementation detail of the fastpfor bitpacking algorithm
223 static constexpr uint64_t CHUNK_SIZE = 32;
224
225public:
226 IntegerBitpacking() = default;
227 IntegerBitpacking(const IntegerBitpacking&) = default;
228
229 void setValuesFromUncompressed(const uint8_t* srcBuffer, common::offset_t srcOffset,
230 uint8_t* dstBuffer, common::offset_t dstOffset, common::offset_t numValues,
231 const CompressionMetadata& metadata, const common::NullMask* nullMask) const final;
232
233 BitpackHeader getBitWidth(const uint8_t* srcBuffer, uint64_t numValues) const;
234
235 static inline uint64_t numValues(uint64_t dataSize, const BitpackHeader& header) {
236 if (header.bitWidth == 0) {
237 return UINT64_MAX;
238 }
239 auto numValues = dataSize * 8 / header.bitWidth;
240 // Round down to nearest multiple of CHUNK_SIZE to ensure that we don't write any extra
241 // values Rounding up could overflow the buffer
242 // TODO(bmwinger): Pack extra values into the space at the end. This will probably be
243 // slower, but only needs to be done once.
244 numValues -= numValues % CHUNK_SIZE;
245 return numValues;
246 }
247
248 CompressionMetadata getCompressionMetadata(const uint8_t* srcBuffer,
249 uint64_t numValues) const override {
250 auto header = getBitWidth(srcBuffer, numValues);
251 // Use uncompressed if the bitwidth is equal to the size of the type in bits
252 if (header.bitWidth >= sizeof(T) * 8) {
253 return CompressionMetadata();
254 }
255 return CompressionMetadata(CompressionType::INTEGER_BITPACKING, header.getData());
256 }

258 uint64_t compressNextPage(const uint8_t*& srcBuffer, uint64_t numValuesRemaining,
259 uint8_t* dstBuffer, uint64_t dstBufferSize,
260 const struct CompressionMetadata& metadata) const final;
261
262 void decompressFromPage(const uint8_t* srcBuffer, uint64_t srcOffset, uint8_t* dstBuffer,
263 uint64_t dstOffset, uint64_t numValues,
264 const struct CompressionMetadata& metadata) const final;
265
266 static bool canUpdateInPlace(T value, const BitpackHeader& header);
267
268protected:
269 // Read multiple values from within a chunk. Cannot span multiple chunks.
270 void getValues(const uint8_t* chunkStart, uint8_t pos, uint8_t* dst, uint8_t numValuesToRead,
271 const BitpackHeader& header) const;
272
273 inline const uint8_t* getChunkStart(const uint8_t* buffer, uint64_t pos,
274 uint8_t bitWidth) const {
275 // Order of operations is important so that pos is rounded down to a multiple of CHUNK_SIZE
276 return buffer + (pos / CHUNK_SIZE) * bitWidth * CHUNK_SIZE / 8;
277 }
