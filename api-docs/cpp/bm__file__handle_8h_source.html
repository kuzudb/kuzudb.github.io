15// Keeps the state information of a page in a file.
16class PageState {
17 static constexpr uint64_t DIRTY_MASK = 0x0080000000000000;
18 static constexpr uint64_t STATE_MASK = 0xFF00000000000000;
19 static constexpr uint64_t VERSION_MASK = 0x00FFFFFFFFFFFFFF;
20 static constexpr uint64_t NUM_BITS_TO_SHIFT_FOR_STATE = 56;
21
22public:
23 static constexpr uint64_t UNLOCKED = 0;
24 static constexpr uint64_t LOCKED = 1;
25 static constexpr uint64_t MARKED = 2;
26 static constexpr uint64_t EVICTED = 3;
27
28 PageState() {
29 stateAndVersion.store(EVICTED << NUM_BITS_TO_SHIFT_FOR_STATE, std::memory_order_release);
30 }
31
32 inline uint64_t getState() { return getState(stateAndVersion.load()); }
33 inline static uint64_t getState(uint64_t stateAndVersion) {
34 return (stateAndVersion & STATE_MASK) >> NUM_BITS_TO_SHIFT_FOR_STATE;
35 }
36 inline static uint64_t getVersion(uint64_t stateAndVersion) {
37 return stateAndVersion & VERSION_MASK;
38 }
39 inline static uint64_t updateStateWithSameVersion(uint64_t oldStateAndVersion,
40 uint64_t newState) {
41 return ((oldStateAndVersion << 8) >> 8) | (newState << NUM_BITS_TO_SHIFT_FOR_STATE);
42 }
43 inline static uint64_t updateStateAndIncrementVersion(uint64_t oldStateAndVersion,
44 uint64_t newState) {
45 return (((oldStateAndVersion << 8) >> 8) + 1) | (newState << NUM_BITS_TO_SHIFT_FOR_STATE);
46 }
47 inline void spinLock(uint64_t oldStateAndVersion) {
48 while (true) {
49 if (tryLock(oldStateAndVersion)) {
50 return;
51 }
52 }
53 }
54 inline bool tryLock(uint64_t oldStateAndVersion) {
55 return stateAndVersion.compare_exchange_strong(oldStateAndVersion,
56 updateStateWithSameVersion(oldStateAndVersion, LOCKED));
57 }
58 inline void unlock() {
59 // TODO(Keenan / Guodong): Track down this rare bug and re-enable the assert. Ref #2289.
60 // KU_ASSERT(getState(stateAndVersion.load()) == LOCKED);
61 stateAndVersion.store(updateStateAndIncrementVersion(stateAndVersion.load(), UNLOCKED),
62 std::memory_order_release);
63 }
64 // Change page state from Mark to Unlocked.
65 inline bool tryClearMark(uint64_t oldStateAndVersion) {
66 KU_ASSERT(getState(oldStateAndVersion) == MARKED);
67 return stateAndVersion.compare_exchange_strong(oldStateAndVersion,
68 updateStateWithSameVersion(oldStateAndVersion, UNLOCKED));
69 }
70 inline bool tryMark(uint64_t oldStateAndVersion) {
71 return stateAndVersion.compare_exchange_strong(oldStateAndVersion,
72 updateStateWithSameVersion(oldStateAndVersion, MARKED));
73 }
74
75 inline void setDirty() {
76 KU_ASSERT(getState(stateAndVersion.load()) == LOCKED);
77 stateAndVersion |= DIRTY_MASK;
78 }
79 inline void clearDirty() {
80 KU_ASSERT(getState(stateAndVersion.load()) == LOCKED);
81 stateAndVersion &= ~DIRTY_MASK;
82 }
83 // Meant to be used when flushing in a single thread.
84 // Should not be used if other threads are modifying the page state
85 inline void clearDirtyWithoutLock() { stateAndVersion &= ~DIRTY_MASK; }
86 inline bool isDirty() const { return stateAndVersion & DIRTY_MASK; }
87 uint64_t getStateAndVersion() const { return stateAndVersion.load(); }
88
89 inline void resetToEvicted() {
90 stateAndVersion.store(EVICTED << NUM_BITS_TO_SHIFT_FOR_STATE, std::memory_order_release);
91 }
92
93private:
94 // Highest 1 bit is dirty bit, and the rest are page state and version bits.
95 // In the rest bits, the lowest 1 byte is state, and the rest are version.
96 std::atomic<uint64_t> stateAndVersion;
97};

99// This class is used to keep the WAL page idxes of a page group in the original file handle.
100// For each original page in the page group, it has a corresponding WAL page idx. The WAL page idx
101// is initialized as INVALID_PAGE_IDX to indicate that the page does not have any updates in WAL
102// file. To synchronize accesses to the WAL version page idxes, we use a lock for each WAL page idx.
103class WALPageIdxGroup {
104public:
105 WALPageIdxGroup();
106
107 // `originalPageIdxInGroup` is the page idx of the original page within the page group.
108 // For example, given a page idx `x` in a file, its page group id is `x / PAGE_GROUP_SIZE`, and
109 // the pageIdxInGroup is `x % PAGE_GROUP_SIZE`.
110 inline void acquireWALPageIdxLock(common::page_idx_t originalPageIdxInGroup) {
111 walPageIdxLocks[originalPageIdxInGroup]->lock();
112 }
113 inline void releaseWALPageIdxLock(common::page_idx_t originalPageIdxInGroup) {
114 walPageIdxLocks[originalPageIdxInGroup]->unlock();
115 }
116 inline common::page_idx_t getWALVersionPageIdxNoLock(common::page_idx_t pageIdxInGroup) const {
117 return walPageIdxes[pageIdxInGroup];
118 }
119 inline void setWALVersionPageIdxNoLock(common::page_idx_t pageIdxInGroup,
120 common::page_idx_t walVersionPageIdx) {
121 walPageIdxes[pageIdxInGroup] = walVersionPageIdx;
122 }
123
124private:
125 std::vector<common::page_idx_t> walPageIdxes;
126 std::vector<std::unique_ptr<std::mutex>> walPageIdxLocks;
127};
